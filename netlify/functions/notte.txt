function doPost(e) {
  var data = {};
  try {
    data = JSON.parse(e.postData.contents);
  } catch (err) {
    data = e.parameter;
  }

  var recipient = "bamigboyeayomide@gmail.com"; // Change to your email
  var subject = "New Waitlist Signup!";
  var name = data.name || "Unknown";
  var email = data.email || "Unknown";
  var count = data.waitlistCount || "Unknown";

  var body = "A new person joined the waitlist!\n\n" +
             "Name: " + name + "\n" +
             "Email: " + email + "\n" +
             "Current waitlist count: " + count;

  MailApp.sendEmail({
    to: recipient,
    subject: subject,
    body: body
  });

  return ContentService.createTextOutput("Success");
